644 template <typename T>
645 struct classify_object<
646  T,
647  typename std::enable_if<is_tuple_like<T>::value &&
648  ((type_count<T>::value >= 2 && !is_wrapper<T>::value) ||
649  (uncommon_type<T>::value && !is_direct_constructible<T, double>::value &&
650  !is_direct_constructible<T, int>::value))>::type> {
651  static constexpr object_category value{object_category::tuple_value};
652  // the condition on this class requires it be like a tuple, but on some compilers (like Xcode) tuples can be

811 inline std::int64_t to_flag_value(std::string val) {
812  static const std::string trueString("true");
813  static const std::string falseString("false");
814  if(val == trueString) {
815  return 1;
816  }
817  if(val == falseString) {
818  return -1;
819  }
820  val = detail::to_lower(val);
821  std::int64_t ret;
822  if(val.size() == 1) {
823  if(val[0] >= '1' && val[0] <= '9') {
824  return (static_cast<std::int64_t>(val[0]) - '0');
825  }
826  switch(val[0]) {
827  case '0':
828  case 'f':
829  case 'n':
830  case '-':
831  ret = -1;
832  break;
833  case 't':
834  case 'y':
835  case '+':
836  ret = 1;
837  break;
838  default:
839  throw std::invalid_argument("unrecognized character");
840  }
841  return ret;
842  }
843  if(val == trueString || val == "on" || val == "yes" || val == "enable") {
844  ret = 1;
845  } else if(val == falseString || val == "off" || val == "no" || val == "disable") {
846  ret = -1;
847  } else {
848  ret = std::stoll(val);
849  }
850  return ret;
851 }
