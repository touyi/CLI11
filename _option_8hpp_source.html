757  std::string get_name(bool positional = false,
758  bool all_options = false
759  ) const {
760  if(get_group().empty())
761  return {}; // Hidden
762 
763  if(all_options) {
764 
765  std::vector<std::string> name_list;
766 
768  if((positional && (!pname_.empty())) || (snames_.empty() && lnames_.empty())) {
769  name_list.push_back(pname_);
770  }
771  if((get_items_expected() == 0) && (!fnames_.empty())) {
772  for(const std::string &sname : snames_) {
773  name_list.push_back("-" + sname);
774  if(check_fname(sname)) {
775  name_list.back() += "{" + get_flag_value(sname, "") + "}";
776  }
777  }
778 
779  for(const std::string &lname : lnames_) {
780  name_list.push_back("--" + lname);
781  if(check_fname(lname)) {
782  name_list.back() += "{" + get_flag_value(lname, "") + "}";
783  }
784  }
785  } else {
786  for(const std::string &sname : snames_)
787  name_list.push_back("-" + sname);
788 
789  for(const std::string &lname : lnames_)
790  name_list.push_back("--" + lname);
791  }
792 
793  return detail::join(name_list);
794  }
795 
796  // This returns the positional name no matter what
797  if(positional)
798  return pname_;
799 
800  // Prefer long name
801  if(!lnames_.empty())
802  return std::string(2, '-') + lnames_[0];
803 
804  // Or short name if no long name
805  if(!snames_.empty())
806  return std::string(1, '-') + snames_[0];
807 
808  // If positional is the only name, it's okay to use that
809  return pname_;
810  }

817  void run_callback() {
818 
819  if(current_option_state_ == option_state::parsing) {
820  _validate_results(results_);
821  current_option_state_ = option_state::validated;
822  }
823 
824  if(current_option_state_ < option_state::reduced) {
825  _reduce_results(proc_results_, results_);
826  current_option_state_ = option_state::reduced;
827  }
828  if(current_option_state_ >= option_state::reduced) {
829  current_option_state_ = option_state::callback_run;
830  if(!(callback_)) {
831  return;
832  }
833  const results_t &send_results = proc_results_.empty() ? results_ : proc_results_;
834  bool local_result = callback_(send_results);
835 
836  if(!local_result)
837  throw ConversionError(get_name(), results_);
838  }
839  }

842  const std::string &matching_name(const Option &other) const {
843  static const std::string estring;
844  for(const std::string &sname : snames_)
845  if(other.check_sname(sname))
846  return sname;
847  for(const std::string &lname : lnames_)
848  if(other.check_lname(lname))
849  return lname;
850 
851  if(ignore_case_ ||
852  ignore_underscore_) { // We need to do the inverse, in case we are ignore_case or ignore underscore
853  for(const std::string &sname : other.snames_)
854  if(check_sname(sname))
855  return sname;
856  for(const std::string &lname : other.lnames_)
857  if(check_lname(lname))
858  return lname;
859  }
860  return estring;
861  }

866  bool check_name(const std::string &name) const {
867 
868  if(name.length() > 2 && name[0] == '-' && name[1] == '-')
869  return check_lname(name.substr(2));
870  if(name.length() > 1 && name.front() == '-')
871  return check_sname(name.substr(1));
872  if(!pname_.empty()) {
873  std::string local_pname = pname_;
874  std::string local_name = name;
875  if(ignore_underscore_) {
876  local_pname = detail::remove_underscore(local_pname);
877  local_name = detail::remove_underscore(local_name);
878  }
879  if(ignore_case_) {
880  local_pname = detail::to_lower(local_pname);
881  local_name = detail::to_lower(local_name);
882  }
883  if(local_name == local_pname) {
884  return true;
885  }
886  }
887 
888  if(!envname_.empty()) {
889  // this needs to be the original since envname_ shouldn't match on case insensitivity
890  return (name == envname_);
891  }
892  return false;
893  }

915  std::string get_flag_value(const std::string &name, std::string input_value) const {
916  static const std::string trueString{"true"};
917  static const std::string falseString{"false"};
918  static const std::string emptyString{"{}"};
919  // check for disable flag override_
920  if(disable_flag_override_) {

1172  if(!validators_.empty()) {
1173  if(type_size_max_ > 1) { // in this context index refers to the index in the type

1211  void _reduce_results(results_t &res, const results_t &original) const {
1212 
1213  // max num items expected or length of vector, always at least 1
1214  // Only valid for a trimming policy
1215 
1216  res.clear();
1217  // Operation depends on the policy setting
1218  switch(multi_option_policy_) {
1219  case MultiOptionPolicy::TakeAll:
1220  break;
1221  case MultiOptionPolicy::TakeLast: {
1222  // Allow multi-option sizes (including 0)
1223  std::size_t trim_size = std::min<std::size_t>(
1224  static_cast<std::size_t>(std::max<int>(get_items_expected_max(), 1)), original.size());
1225  if(original.size() != trim_size) {
1226  res.assign(original.end() - static_cast<results_t::difference_type>(trim_size), original.end());
1227  }
1228  } break;
1229  case MultiOptionPolicy::TakeFirst: {
1230  std::size_t trim_size = std::min<std::size_t>(
1231  static_cast<std::size_t>(std::max<int>(get_items_expected_max(), 1)), original.size());
1232  if(original.size() != trim_size) {
1233  res.assign(original.begin(), original.begin() + static_cast<results_t::difference_type>(trim_size));
1234  }
1235  } break;
1236  case MultiOptionPolicy::Join:
1237  if(results_.size() > 1) {
1238  res.push_back(detail::join(original, std::string(1, (delimiter_ == '\0') ? '\n' : delimiter_)));
1239  }
1240  break;
1241  case MultiOptionPolicy::Throw:
1242  default: {
1243  auto num_min = static_cast<std::size_t>(get_items_expected_min());
1244  auto num_max = static_cast<std::size_t>(get_items_expected_max());
1245  if(num_min == 0) {
1246  num_min = 1;
1247  }
1248  if(num_max == 0) {
1249  num_max = 1;
1250  }
1251  if(original.size() < num_min) {
1252  throw ArgumentMismatch::AtLeast(get_name(), static_cast<int>(num_min), original.size());
1253  }
1254  if(original.size() > num_max) {
1255  throw ArgumentMismatch::AtMost(get_name(), static_cast<int>(num_max), original.size());
1256  }
1257  break;
1258  }
1259  }
1260  }
