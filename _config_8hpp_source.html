24 inline std::string convert_arg_for_ini(const std::string &arg) {
25  if(arg.empty()) {
26  return std::string(2, '"');
27  }
28  // some specifically supported strings
29  if(arg == "true" || arg == "false" || arg == "nan" || arg == "inf") {
30  return arg;
31  }
32  // floating point conversion can convert some hex codes, but don't try that here
33  if(arg.compare(0, 2, "0x") != 0 && arg.compare(0, 2, "0X") != 0) {
34  double val;
35  if(detail::lexical_cast(arg, val)) {
36  return arg;
37  }
38  }
39  // just quote a single non numeric character
40  if(arg.size() == 1) {
41  return std::string("'") + arg + '\'';
42  }
43  // handle hex, binary or octal arguments
44  if(arg.front() == '0') {
45  if(arg[1] == 'x') {
46  if(std::all_of(arg.begin() + 2, arg.end(), [](char x) {
47  return (x >= '0' && x <= '9') || (x >= 'A' && x <= 'F') || (x >= 'a' && x <= 'f');
48  })) {
49  return arg;
50  }
51  } else if(arg[1] == 'o') {
52  if(std::all_of(arg.begin() + 2, arg.end(), [](char x) { return (x >= '0' && x <= '7'); })) {
53  return arg;
54  }
55  } else if(arg[1] == 'b') {
56  if(std::all_of(arg.begin() + 2, arg.end(), [](char x) { return (x == '0' || x == '1'); })) {
57  return arg;
58  }
59  }
60  }
61  if(arg.find_first_of('"') == std::string::npos) {
62  return std::string("\"") + arg + '"';
63  } else {
64  return std::string("'") + arg + '\'';
65  }
66 }

69 inline std::string
70 ini_join(const std::vector<std::string> &args, char sepChar = ',', char arrayStart = '[', char arrayEnd = ']') {
71  std::string joined;
72  if(args.size() > 1 && arrayStart != '\0') {
73  joined.push_back(arrayStart);
74  }
75  std::size_t start = 0;
76  for(const auto &arg : args) {
77  if(start++ > 0) {
78  joined.push_back(sepChar);
79  if(isspace(sepChar) == 0) {
80  joined.push_back(' ');
81  }
82  }
83  joined.append(convert_arg_for_ini(arg));
84  }
85  if(args.size() > 1 && arrayEnd != '\0') {
86  joined.push_back(arrayEnd);
87  }
88  return joined;
89 }

91 inline std::vector<std::string> generate_parents(const std::string &section, std::string &name) {
92  std::vector<std::string> parents;
93  if(detail::to_lower(section) != "default") {
94  if(section.find('.') != std::string::npos) {
95  parents = detail::split(section, '.');
96  } else {
97  parents = {section};
98  }
99  }
100  if(name.find('.') != std::string::npos) {
101  std::vector<std::string> plist = detail::split(name, '.');
102  name = plist.back();
103  detail::remove_quotes(name);
104  plist.pop_back();
105  parents.insert(parents.end(), plist.begin(), plist.end());
106  }
107 
108  // clean up quotes on the parents
109  for(auto &parent : parents) {
110  detail::remove_quotes(parent);
111  }
112  return parents;
113 }

116 inline void checkParentSegments(std::vector<ConfigItem> &output, const std::string &currentSection) {
117 
118  std::string estring;
119  auto parents = detail::generate_parents(currentSection, estring);
120  if(!output.empty() && output.back().name == "--") {
121  std::size_t msize = (parents.size() > 1U) ? parents.size() : 2;
122  while(output.back().parents.size() >= msize) {
123  output.push_back(output.back());
124  output.back().parents.pop_back();
125  }
126 
127  if(parents.size() > 1) {
128  std::size_t common = 0;
129  std::size_t mpair = (std::min)(output.back().parents.size(), parents.size() - 1);
130  for(std::size_t ii = 0; ii < mpair; ++ii) {
131  if(output.back().parents[ii] != parents[ii]) {
132  break;
133  }
134  ++common;
135  }
136  if(common == mpair) {
137  output.pop_back();
138  } else {
139  while(output.back().parents.size() > common + 1) {
140  output.push_back(output.back());
141  output.back().parents.pop_back();
142  }
143  }
144  for(std::size_t ii = common; ii < parents.size() - 1; ++ii) {
145  output.emplace_back();
146  output.back().parents.assign(parents.begin(), parents.begin() + static_cast<std::ptrdiff_t>(ii) + 1);
147  output.back().name = "++";
148  }
149  }
150  } else if(parents.size() > 1) {
151  for(std::size_t ii = 0; ii < parents.size() - 1; ++ii) {
152  output.emplace_back();
153  output.back().parents.assign(parents.begin(), parents.begin() + static_cast<std::ptrdiff_t>(ii) + 1);
154  output.back().name = "++";
155  }
156  }
157 
158  // insert a section end which is just an empty items_buffer
159  output.emplace_back();
160  output.back().parents = std::move(parents);
161  output.back().name = "++";
162 }

165 inline std::vector<ConfigItem> ConfigBase::from_config(std::istream &input) const {
166  std::string line;
167  std::string section = "default";
168 
169  std::vector<ConfigItem> output;
170  bool isDefaultArray = (arrayStart == '[' && arrayEnd == ']' && arraySeparator == ',');
171  bool isINIArray = (arrayStart == '\0' || arrayStart == ' ') && arrayStart == arrayEnd;
172  char aStart = (isINIArray) ? '[' : arrayStart;
173  char aEnd = (isINIArray) ? ']' : arrayEnd;
174  char aSep = (isINIArray && arraySeparator == ' ') ? ',' : arraySeparator;
175 
176  while(getline(input, line)) {
177  std::vector<std::string> items_buffer;
178  std::string name;
179 
180  detail::trim(line);
181  std::size_t len = line.length();
182  if(len > 1 && line.front() == '[' && line.back() == ']') {
183  if(section != "default") {
184  // insert a section end which is just an empty items_buffer
185  output.emplace_back();
186  output.back().parents = detail::generate_parents(section, name);
187  output.back().name = "--";
188  }
189  section = line.substr(1, len - 2);
190  // deal with double brackets for TOML
191  if(section.size() > 1 && section.front() == '[' && section.back() == ']') {
192  section = section.substr(1, section.size() - 2);
193  }
194  if(detail::to_lower(section) == "default") {
195  section = "default";
196  } else {
197  detail::checkParentSegments(output, section);
198  }
199  continue;
200  }
201  if(len == 0) {
202  continue;
203  }
204  // comment lines
205  if(line.front() == ';' || line.front() == '#' || line.front() == commentChar) {
206  continue;
207  }
208 
209  // Find = in string, split and recombine
210  auto pos = line.find(valueDelimiter);
211  if(pos != std::string::npos) {
212  name = detail::trim_copy(line.substr(0, pos));
213  std::string item = detail::trim_copy(line.substr(pos + 1));
214  if(item.size() > 1 && item.front() == aStart) {
215  for(std::string multiline; item.back() != aEnd && std::getline(input, multiline);) {
216  detail::trim(multiline);
217  item += multiline;
218  }
219  items_buffer = detail::split_up(item.substr(1, item.length() - 2), aSep);
220  } else if((isDefaultArray || isINIArray) && item.find_first_of(aSep) != std::string::npos) {
221  items_buffer = detail::split_up(item, aSep);
222  } else if((isDefaultArray || isINIArray) && item.find_first_of(' ') != std::string::npos) {
223  items_buffer = detail::split_up(item);
224  } else {
225  items_buffer = {item};
226  }
227  } else {
228  name = detail::trim_copy(line);
229  items_buffer = {"true"};
230  }
231  if(name.find('.') == std::string::npos) {
232  detail::remove_quotes(name);
233  }
234  // clean up quotes on the items
235  for(auto &it : items_buffer) {
236  detail::remove_quotes(it);
237  }
238 
239  std::vector<std::string> parents = detail::generate_parents(section, name);
240 
241  if(!output.empty() && name == output.back().name && parents == output.back().parents) {
242  output.back().inputs.insert(output.back().inputs.end(), items_buffer.begin(), items_buffer.end());
243  } else {
244  output.emplace_back();
245  output.back().parents = std::move(parents);
246  output.back().name = std::move(name);
247  output.back().inputs = std::move(items_buffer);
248  }
249  }
250  if(section != "default") {
251  // insert a section end which is just an empty items_buffer
252  std::string ename;
253  output.emplace_back();
254  output.back().parents = detail::generate_parents(section, ename);
255  output.back().name = "--";
256  while(output.back().parents.size() > 1) {
257  output.push_back(output.back());
258  output.back().parents.pop_back();
259  }
260  }
261  return output;
262 }

264 inline std::string
265 ConfigBase::to_config(const App *app, bool default_also, bool write_description, std::string prefix) const {
266  std::stringstream out;
267  std::string commentLead;
268  commentLead.push_back(commentChar);
269  commentLead.push_back(' ');
270 
271  std::vector<std::string> groups = app->get_groups();
272  bool defaultUsed = false;
273  groups.insert(groups.begin(), std::string("Options"));
274  if(write_description) {
275  out << commentLead << app->get_description() << '\n';
276  }
277  for(auto &group : groups) {
278  if(group == "Options" || group.empty()) {
279  if(defaultUsed) {
280  continue;
281  }
282  defaultUsed = true;
283  }
284  if(write_description && group != "Options" && !group.empty()) {
285  out << '\n' << commentLead << group << " Options\n";
286  }
287  for(const Option *opt : app->get_options({})) {
288 
289  // Only process options that are configurable
290  if(opt->get_configurable()) {
291  if(opt->get_group() != group) {
292  if(!(group == "Options" && opt->get_group().empty())) {
293  continue;
294  }
295  }
296  std::string name = prefix + opt->get_single_name();
297  std::string value = detail::ini_join(opt->reduced_results(), arraySeparator, arrayStart, arrayEnd);
298 
299  if(value.empty() && default_also) {
300  if(!opt->get_default_str().empty()) {
301  value = detail::convert_arg_for_ini(opt->get_default_str());
302  } else if(opt->get_expected_min() == 0) {
303  value = "false";
304  }
305  }
306 
307  if(!value.empty()) {
308  if(write_description && opt->has_description()) {
309  out << '\n';
310  out << commentLead << detail::fix_newlines(commentLead, opt->get_description()) << '\n';
311  }
312  out << name << valueDelimiter << value << '\n';
313  }
314  }
315  }
316  }
317  auto subcommands = app->get_subcommands({});
318  for(const App *subcom : subcommands) {
319  if(subcom->get_name().empty()) {
320  if(write_description && !subcom->get_group().empty()) {
321  out << '\n' << commentLead << subcom->get_group() << " Options\n";
322  }
323  out << to_config(subcom, default_also, write_description, prefix);
324  }
325  }
326 
327  for(const App *subcom : subcommands) {
328  if(!subcom->get_name().empty()) {
329  if(subcom->get_configurable() && app->got_subcommand(subcom)) {
330  if(!prefix.empty() || app->get_parent() == nullptr) {
331  out << '[' << prefix << subcom->get_name() << "]\n";
332  } else {
333  std::string subname = app->get_name() + "." + subcom->get_name();
334  auto p = app->get_parent();
335  while(p->get_parent() != nullptr) {
336  subname = p->get_name() + "." + subname;
337  p = p->get_parent();
338  }
339  out << '[' << subname << "]\n";
340  }
341  out << to_config(subcom, default_also, write_description, "");
342  } else {
343  out << to_config(subcom, default_also, write_description, prefix + subcom->get_name() + ".");
344  }
345  }
346  }
347 
348  return out.str();
349 }
